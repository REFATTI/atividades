#include <stdio.h>


int saoMultiplos(int a, int b) {
    return (a % b == 0 || b % a == 0);
}

int main() {
    int a, b;

    
    scanf("%d %d", &a, &b);

   
    if (saoMultiplos(a, b)) {
        printf("Sao Multiplos\n");
    } else {
        printf("Nao sao Multiplos\n");
    }

    return 0;
}
